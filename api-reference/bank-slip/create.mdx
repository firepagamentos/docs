async function checkBoletoStatus(boletoId) {
  try {
    const response = await fetch(`https://api-gateway.firebanking.com.br/bank-slip/v1/payment/${boletoId}`, {
      headers: {

async function validateAndFillAddress(cep) {
  const cleanCep = cep.replace(/\D/g, '');
  
  if (cleanCep.length !== 8) {
    throw new Error('CEP deve ter 8 dﾃｭgitos');
  }
  
  try {
    const response = await fetch(`https://viacep.com.br/ws/${cleanCep}/json/`);
    const address = await response.json();
    
    if (address.erro) {
      throw new Error('CEP nﾃ｣o encontrado');
    }
    
    return {
      street: address.logradouro,
      district: address.bairro,
      city: address.localidade,
      state: address.uf,

    };
  } catch (error) {
    throw new Error('Erro ao consultar CEP: ' + error.message);
  }
}

## Boas Prﾃ｡ticas

### Dados do Pagador
- **Valide CPF/CNPJ**: Use algoritmos de validaﾃｧﾃ｣o antes de enviar
- **CEP obrigatﾃｳrio**: Endereﾃｧo completo ﾃｩ exigido pelos bancos
- **Email vﾃ｡lido**: Necessﾃ｡rio para envio automﾃ｡tico do boleto
- **Nome completo**: Evite abreviaﾃｧﾃｵes ou apelidos

### Gestﾃ｣o de Vencimentos
- **Prazo adequado**: 7-15 dias para e-commerce, 30 dias para B2B
- **Considere fins de semana**: Evite vencimento em sﾃ｡bados/domingos
- **Planeje feriados**: Considere feriados nacionais e locais
- **Margem de seguranﾃｧa**: Dﾃｪ tempo para processamento bancﾃ｡rio

### Experiﾃｪncia do Cliente
- **Instruﾃｧﾃｵes claras**: Explique como pagar o boleto
- **Mﾃｺltiplos formatos**: Ofereﾃｧa PDF, cﾃｳdigo de barras e linha digitﾃ｡vel
- **Compartilhamento fﾃ｡cil**: Facilite envio por email/WhatsApp
- **Status em tempo real**: Informe quando o pagamento foi confirmado
